h2. adlist
adlist是个基本的双端链表。支持自定义dup、match方法，用于复制list和在list中搜索，其他部分倒是没有什么特别的。
{% highlight c %}
typedef struct list {
    listNode *head;
    listNode *tail;
    void *(*dup)(void *ptr);
    void (*free)(void *ptr);
    int (*match)(void *ptr, void *key);
    unsigned long len;
} list;
{% endhighlight %}
